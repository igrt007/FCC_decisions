class Planet:
    def __init__(self, name, planet_type, star):
        self.name = name
        self.planet_type = planet_type
        self.star = star

        if not isinstance(self.name, str):
            raise TypeError("name, planet type, and star must be strings")
        elif not isinstance(self.planet_type, str):
            raise TypeError("name, planet type, and star must be strings")
        elif not isinstance(self.star, str):
            raise TypeError("name, planet type, and star must be strings")
        
        if len(self.name) == 0:
            raise ValueError("name, planet_type, and star must be non-empty strings")
        elif len(self.planet_type) == 0:
            raise ValueError("name, planet_type, and star must be non-empty strings")
        elif len(self.star) == 0:
            raise ValueError("name, planet_type, and star must be non-empty strings")

    def orbit(self):
        return(f"{self.name} is orbiting around {self.star}...")
    def __str__(self):
        return(f"Planet: {self.name} | Type: {self.planet_type} | Star: {self.star}")

planet_1 = Planet('Tera', 'Earth', 'Sun1')
planet_2 = Planet('Jupiter', 'Mega-earth', 'Sun2')
planet_3 = Planet('Mercury', 'sub-earth', 'Sun3')

print(planet_1)
print(planet_2)
print(planet_3)

print(planet_1.orbit())
print(planet_2.orbit())
print(planet_3.orbit())
